```python
import json
import os
import subprocess
import time

def read_json_file(file_path):
    """
    读取JSON文件并返回title和url的列表
    """
    try:
        with open(file_path, 'r', encoding='utf-8') as f:
            data = json.load(f)
        # 确保数据格式正确
        download_list = []
        if isinstance(data, list):
            for item in data:
                if 'title' in item and 'url' in item:
                    download_list.append({
                        'title': item['title'],
                        'url': item['url']
                    })
        elif isinstance(data, dict) and 'title' in data and 'url' in data:
            download_list.append({
                'title': data['title'],
                'url': data['url']
            })
        return download_list
    except Exception as e:
        print(f"读取JSON文件失败: {e}")
        return []

def download_with_idm(url, save_path, file_name, idm_path=r"C:\Program Files (x86)\Internet Download Manager\IDMan.exe"):
    """
    使用IDM下载文件（仅提交任务，不等待完成）
    :param url: 下载链接
    :param save_path: 保存路径
    :param file_name: 文件名
    :param idm_path: IDM安装路径
    :return: 是否提交下载成功
    """
    # 检查IDM是否存在
    if not os.path.exists(idm_path):
        print(f"IDM未找到，请检查路径: {idm_path}")
        return False
    
    # 创建保存目录
    os.makedirs(save_path, exist_ok=True)
    
    # IDM命令行参数说明：
    # /d - 下载链接
    # /p - 保存路径
    # /f - 文件名
    # /n - 不询问用户，直接开始下载
    # /s - 静默模式下载
    cmd = [
        idm_path,
        '/d', url,
        '/p', save_path,
        '/f', file_name,
        '/n', '/s'
    ]
    
    try:
        # 执行IDM命令
        subprocess.run(cmd, check=True, capture_output=True, text=True)
        print(f"✅ 已提交下载任务: {file_name}")
        return True
    except subprocess.CalledProcessError as e:
        print(f"❌ 下载任务提交失败: {file_name}, 错误: {e}")
        return False

def main(json_file_path):
    """
    主函数：读取JSON，批量提交IDM下载任务
    """
    # 固定下载路径（你指定的路径）
    DOWNLOAD_DIR = r"C:\Users\31184\Desktop\gxq"
    
    # 1. 读取JSON文件
    download_items = read_json_file(json_file_path)
    if not download_items:
        print("⚠️ 没有可下载的内容")
        return
    
    # 2. 批量提交下载任务
    success_count = 0
    fail_count = 0
    
    for item in download_items:
        title = item['title']
        url = item['url']
        
        # 清理文件名中的非法字符（避免IDM报错）
        safe_title = "".join([c for c in title if c not in r'\/:*?"<>|'])
        # 获取文件扩展名（从URL中提取，保证文件格式正确）
        file_ext = os.path.splitext(url)[1] if os.path.splitext(url)[1] else ''
        # 最终文件名 = 清理后的title + 扩展名
        final_filename = f"{safe_title}{file_ext}"
        
        # 提交IDM下载任务
        if download_with_idm(url, DOWNLOAD_DIR, final_filename):
            success_count += 1
        else:
            fail_count += 1
        
        # 短暂延迟，避免IDM处理不过来
        time.sleep(0.5)
    
    # 输出统计结果
    print("\n" + "-" * 60)
    print(f"📊 任务提交完成！")
    print(f"✅ 成功提交: {success_count} 个任务")
    print(f"❌ 提交失败: {fail_count} 个任务")
    print(f"📁 下载文件将保存到: {DOWNLOAD_DIR}")

if __name__ == "__main__":
    # 配置JSON文件路径
    JSON_FILE = "download_list.json"  # 你的JSON文件路径
    
    # 运行主程序
    main(JSON_FILE)
```
